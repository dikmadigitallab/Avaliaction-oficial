generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
}

enum UserType {
  EMPRESA
  GERENTE
  SUPERVISOR
  ADMINISTRATOR
  ADMIN
}

enum QuestionType {
  TEXT
  AVALIACAO
  BOOLEAN
  CHECKBOX
  RADIO
}

model User {
  id          String   @id @default(uuid())
  userType    UserType @default(GERENTE)
  nome        String
  email       String   @unique
  cpf         String   @unique
  senha       String

  cpfs        Cpf[]
  formularios Form[]

  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

model Cpf {
  id        String   @id @default(uuid())
  cpf       String   @unique

  userId    String?
  user      User?    @relation(fields: [userId], references: [id])

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

model Form {
  id        String     @id @default(uuid())
  name      String
  anonymous Boolean    @default(true)

  userId    String
  user      User       @relation(fields: [userId], references: [id])

  questions Question[]
  respostas Resposta[]

  createdAt DateTime   @default(now())
  updatedAt DateTime   @updatedAt
}

model Question {
  id        String       @id @default(uuid())
  pergunta  String
  type      QuestionType @default(TEXT)
  required  Boolean      @default(false)
  order     Int

  formId    String
  form      Form         @relation(fields: [formId], references: [id])

  createdAt DateTime     @default(now())
  updatedAt DateTime     @updatedAt
}

model Resposta {
  id        String   @id @default(cuid())

  formId    String
  form      Form     @relation(fields: [formId], references: [id])

  createdAt DateTime @default(now())
}
